75  	// Initialize numCPU and open a Performance Data Helper Thread for
76  	// monitoring each processor ticks
77  	private static PointerByReference phQuery = new PointerByReference();
78  	private static final IntByReference zero = new IntByReference(0);
79  	private static final int numCPU;
80  	// Set up Performance Data Helper thread for uptime
81  	private static PointerByReference uptimeQuery = new PointerByReference();
82  	private static final IntByReference one = new IntByReference(1);
83  
84  	static {
85  		// Get number of processors
86  		SYSTEM_INFO sysinfo = new SYSTEM_INFO();
87  		Kernel32.INSTANCE.GetSystemInfo(sysinfo);
88  		numCPU = sysinfo.dwNumberOfProcessors.intValue();
89  
90  		// Open tick query for this processor
91  		int ret = Pdh.INSTANCE.PdhOpenQuery(null, zero, phQuery);
92  		if (ret != 0)
93  			throw new LastErrorException("Cannot open PDH query. Error code: " + String.format("0x%08X", ret));
94  
95  		// Open uptime query for this processor
96  		ret = Pdh.INSTANCE.PdhOpenQuery(null, one, uptimeQuery);
97  		if (ret != 0)
98  			throw new LastErrorException("Cannot open PDH query. Error code: " + String.format("0x%08X", ret));
99  
100 		// Set up hook to close the query on shutdown
101 		Runtime.getRuntime().addShutdownHook(new Thread() {
102 			@Override
103 			public void run() {
104 				Pdh.INSTANCE.PdhCloseQuery(phQuery.getValue());
105 				Pdh.INSTANCE.PdhCloseQuery(uptimeQuery.getValue());
106 			}
107 		});
108 	}

110 	// Set up a counter for each processor
111 	private static PointerByReference[] phUserCounters = new PointerByReference[numCPU];
112 	private static PointerByReference[] phIdleCounters = new PointerByReference[numCPU];
113 
114 	static {
115 		for (int p = 0; p < numCPU; p++) {
116 			// Options are (only need 2 to calculate all)
117 			// "\Processor(0)\% processor time"
118 			// "\Processor(0)\% idle time" (1 - processor)
119 			// "\Processor(0)\% privileged time" (subset of processor)
120 			// "\Processor(0)\% user time" (other subset of processor)
121 			// Note need to make \ = \\ for Java Strings and %% for format
122 			String counterPath = String.format("\\Processor(%d)\\%% user time", p);
123 			phUserCounters[p] = new PointerByReference();
124 			int ret = Pdh.INSTANCE.PdhAddEnglishCounterA(phQuery.getValue(), counterPath, zero, phUserCounters[p]);
125 			if (ret != 0)
126 				throw new LastErrorException("Cannot add PDH Counter for % user time for processor " + p
127 						+ ". Error code: " + String.format("0x%08X", ret));
128 			counterPath = String.format("\\Processor(%d)\\%% idle time", p);
129 			phIdleCounters[p] = new PointerByReference();
130 			ret = Pdh.INSTANCE.PdhAddEnglishCounterA(phQuery.getValue(), counterPath, zero, phIdleCounters[p]);
131 			if (ret != 0)
132 				throw new LastErrorException("Cannot add PDH Counter for % idle time for processor " + p
133 						+ ". Error code: " + String.format("0x%08X", ret));
134 		}
135 		// Initialize by collecting data the first time
136 		int ret = Pdh.INSTANCE.PdhCollectQueryData(phQuery.getValue());
137 		if (ret != 0)
138 			throw new LastErrorException("Cannot collect PDH query data. Error code: " + String.format("0x%08X", ret));
139 	}

166 	/**
167 	 * Create a Processor with the given number
168 	 * 
169 	 * @param procNo
170 	 *            The processor number
171 	 */
172 	public CentralProcessor(int procNo) {
173 		if (procNo >= numCPU)
174 			throw new IllegalArgumentException("Processor number (" + procNo
175 					+ ") must be less than the number of CPUs: " + numCPU);
176 		this.processorNumber = procNo;
177 		updateProcessorTicks();
178 		System.arraycopy(allProcessorTicks[processorNumber], 0, curProcTicks, 0, curProcTicks.length);
179 	}

232 	 * Vendor frequency (in Hz), eg. for processor named Intel(R) Core(TM)2 Duo
233 	 * CPU T7300 @ 2.00GHz the vendor frequency is 2000000000.
234 	 * 
235 	 * @return Processor frequency or -1 if unknown.
236 	 */
237 	@Override
238 	public long getVendorFreq() {
239 		if (this.cpuVendorFreq == null) {
240 			Pattern pattern = Pattern.compile("@ (.*)$");
241 			Matcher matcher = pattern.matcher(getName());
242 
243 			if (matcher.find()) {
244 				String unit = matcher.group(1);
245 				this.cpuVendorFreq = Long.valueOf(ParseUtil.parseHertz(unit));
246 			} else {
247 				this.cpuVendorFreq = Long.valueOf(-1L);
248 			}
249 		}
250 
251 		return this.cpuVendorFreq.longValue();
252 	}

360 	/**
361 	 * {@inheritDoc}
362 	 */
363 	@Override
364 	public double getSystemCpuLoadBetweenTicks() {
365 		// Check if > ~ 0.95 seconds since last tick count.
366 		long now = System.currentTimeMillis();
367 		boolean update = (now - tickTime > 950);
368 		if (update) {
369 			// Enough time has elapsed.
370 			// Update latest
371 			updateSystemTicks();
372 			tickTime = now;
373 		}
374 		// Calculate total
375 		long total = 0;
376 		for (int i = 0; i < curTicks.length; i++) {
377 			total += (curTicks[i] - prevTicks[i]);
378 		}
379 		// Calculate idle from last field [3]
380 		long idle = curTicks[3] - prevTicks[3];
381 
382 		// Copy latest ticks to earlier position for next call
383 		if (update) {
384 			System.arraycopy(curTicks, 0, prevTicks, 0, curTicks.length);
385 		}
386 
387 		// return
388 		if (total > 0 && idle >= 0) {
389 			return (double) (total - idle) / total;
390 		}
391 		return 0d;
392 	}

406 	/**
407 	 * Updates system tick information from native call to GetSystemTimes().
408 	 * Array with four elements representing clock ticks or milliseconds
409 	 * (platform dependent) spent in User (0), Nice (1), System (2), and Idle
410 	 * (3) states. By measuring the difference between ticks across a time
411 	 * interval, CPU load over that interval may be calculated.
412 	 */
413 	private static void updateSystemTicks() {
414 		WinBase.FILETIME lpIdleTime = new WinBase.FILETIME();
415 		WinBase.FILETIME lpKernelTime = new WinBase.FILETIME();
416 		WinBase.FILETIME lpUserTime = new WinBase.FILETIME();
417 		if (0 == Kernel32.INSTANCE.GetSystemTimes(lpIdleTime, lpKernelTime, lpUserTime))
418 			throw new LastErrorException("Error code: " + Native.getLastError());
419 		// Array order is user,nice,kernel,idle
420 		curTicks[0] = lpUserTime.toLong() + Kernel32.WIN32_TIME_OFFSET;
421 		curTicks[1] = 0L; // Windows is not 'nice'
422 		curTicks[2] = lpKernelTime.toLong() - lpIdleTime.toLong();
423 		curTicks[3] = lpIdleTime.toLong() + Kernel32.WIN32_TIME_OFFSET;
424 	}

446 	/**
447 	 * {@inheritDoc}
448 	 */
449 	@Override
450 	public double getProcessorCpuLoadBetweenTicks() {
451 		// Check if > ~ 0.95 seconds since last tick count.
452 		long now = System.currentTimeMillis();
453 		if (now - procTickTime > 950) {
454 			// Enough time has elapsed. Update array in place
455 			updateProcessorTicks();
456 			// Copy arrays in place
457 			System.arraycopy(curProcTicks, 0, prevProcTicks, 0, curProcTicks.length);
458 			System.arraycopy(allProcessorTicks[processorNumber], 0, curProcTicks, 0, curProcTicks.length);
459 			procTickTime = now;
460 		}
461 		long total = 0;
462 		for (int i = 0; i < curProcTicks.length; i++) {
463 			total += (curProcTicks[i] - prevProcTicks[i]);
464 		}
465 		// Calculate idle from last field [3]
466 		long idle = curProcTicks[3] - prevProcTicks[3];
467 		// update
468 		return (total > 0 && idle >= 0) ? (double) (total - idle) / total : 0d;
469 	}

479 	/**
480 	 * Updates the tick array for all processors if more than 100ms has elapsed
481 	 * since the last update. This permits using the allProcessorTicks as a
482 	 * cache when iterating over processors since pdh query updates all counters
483 	 * at once so we can't separate individual processors
484 	 */
485 	private void updateProcessorTicks() {
486 		// Update no more frequently than 100ms so this is only triggered once
487 		// during iteration over Processors
488 		long now = System.currentTimeMillis();
489 		if (now - allProcTickTime < 100)
490 			return;
491 
492 		// This call updates all process counters to % used since last call
493 		int ret = Pdh.INSTANCE.PdhCollectQueryData(phQuery.getValue());
494 		if (ret != 0)
495 			throw new LastErrorException("Cannot collect PDH query data. Error code: " + String.format("0x%08X", ret));
496 		// Multiply % usage times elapsed MS to recreate ticks, then increment
497 		long elapsed = now - allProcTickTime;
498 		for (int cpu = 0; cpu < numCPU; cpu++) {
499 			PdhFmtCounterValue phUserCounterValue = new PdhFmtCounterValue();
500 			ret = Pdh.INSTANCE.PdhGetFormattedCounterValue(phUserCounters[cpu].getValue(), Pdh.PDH_FMT_LARGE
501 					| Pdh.PDH_FMT_1000, null, phUserCounterValue);
502 			if (ret != 0)
503 				throw new LastErrorException("Cannot get PDH User % counter value. Error code: "
504 						+ String.format("0x%08X", ret));
505 
506 			PdhFmtCounterValue phIdleCounterValue = new PdhFmtCounterValue();
507 			ret = Pdh.INSTANCE.PdhGetFormattedCounterValue(phIdleCounters[cpu].getValue(), Pdh.PDH_FMT_LARGE
508 					| Pdh.PDH_FMT_1000, null, phIdleCounterValue);
509 			if (ret != 0)
510 				throw new LastErrorException("Cannot get PDH Idle % counter value. Error code: "
511 						+ String.format("0x%08X", ret));
512 
513 			// Returns results in 1000's of percent, e.g. 5% is 5000
514 			// Multiply by elapsed to get total ms and Divide by 100 * 1000
515 			// Putting division at end avoids need to cast division to double
516 			long user = elapsed * phUserCounterValue.value.largeValue / 100000;
517 			long idle = elapsed * phIdleCounterValue.value.largeValue / 100000;
518 			// Elasped is only since last read, so increment previous value
519 			allProcessorTicks[cpu][0] += user;
520 			// allProcessorTicks[cpu][1] is ignored, Windows is not nice
521 			allProcessorTicks[cpu][2] += (elapsed - user - idle); // u+i+sys=100%
522 			allProcessorTicks[cpu][3] += idle;
523 		}
524 		allProcTickTime = now;
525 	}

527 	/**
528 	 * {@inheritDoc}
529 	 */
530 	public long getSystemUptime() {
531 		int ret = Pdh.INSTANCE.PdhCollectQueryData(uptimeQuery.getValue());
532 		if (ret != 0)
533 			throw new LastErrorException("Cannot collect uptime query data. Error code: "
534 					+ String.format("0x%08X", ret));
535 
536 		PdhFmtCounterValue uptimeCounterValue = new PdhFmtCounterValue();
537 		ret = Pdh.INSTANCE.PdhGetFormattedCounterValue(pUptime.getValue(), Pdh.PDH_FMT_LARGE, null, uptimeCounterValue);
538 		if (ret != 0)
539 			throw new LastErrorException("Cannot get PDH User % counter value. Error code: "
540 					+ String.format("0x%08X", ret));
541 
542 		return uptimeCounterValue.value.largeValue;
543 	}
