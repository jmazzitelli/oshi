17  package oshi.software.os.windows.nt;
18  
19  import java.util.Arrays;
20  import java.util.List;
21  
22  import com.sun.jna.Library;
23  import com.sun.jna.Native;
24  import com.sun.jna.NativeLong;
25  import com.sun.jna.Pointer;
26  import com.sun.jna.Structure;
27  
28  /**
29   * Power profile stats
30   * 
31   * @author widdis[at]gmail[dot]com
32   */
33  public interface PowrProf extends Library {
34  	PowrProf INSTANCE = (PowrProf) Native.loadLibrary("PowrProf", PowrProf.class);
35  
36  	public static int SYSTEM_BATTERY_STATE = 5;
37  
38  	public static class SystemBatteryState extends Structure {
39  		public byte acOnLine; // boolean
40  		public byte batteryPresent; // boolean
41  		public byte charging; // boolean
42  		public byte discharging; // boolean
43  		public byte[] spare1 = new byte[4]; // unused
44  		public int maxCapacity; // unsigned 32 bit
45  		public int remainingCapacity; // unsigned 32 bit
46  		public int rate; // signed 32 bit
47  		public int estimatedTime; // signed 32 bit
48  		public int defaultAlert1; // unsigned 32 bit
49  		public int defaultAlert2; // unsigned 32 bit
50  
51  		@Override
52  		protected List<String> getFieldOrder() {
53  			return Arrays.asList(new String[] { "acOnLine", "batteryPresent", "charging", "discharging", "spare1",
54  					"maxCapacity", "remainingCapacity", "rate", "estimatedTime", "defaultAlert1", "defaultAlert2" });
55  		}
56  	}
57  
58  	int CallNtPowerInformation(int informationLevel, Pointer lpInputBuffer, NativeLong nInputBufferSize,
59  			Structure lpOutputBuffer, NativeLong nOutputBufferSize);
60  }
