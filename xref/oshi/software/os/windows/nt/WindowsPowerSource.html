17  package oshi.software.os.windows.nt;
18  
19  import oshi.hardware.PowerSource;
20  import oshi.software.os.windows.nt.PowrProf.SystemBatteryState;
21  import oshi.util.FormatUtil;
22  
23  import com.sun.jna.NativeLong;
24  
25  /**
26   * A Power Source
27   * 
28   * @author widdis[at]gmail[dot]com
29   */
30  public class WindowsPowerSource implements PowerSource {
31  
32  	private String name;
33  
34  	private double remainingCapacity;
35  
36  	private double timeRemaining;
37  
38  	public WindowsPowerSource(String name, double remainingCapacity,
39  			double timeRemaining) {
40  		this.name = name;
41  		this.remainingCapacity = remainingCapacity;
42  		this.timeRemaining = timeRemaining;
43  	}
44  
45  	@Override
46  	public String getName() {
47  		return this.name;
48  	}
49  
50  	@Override
51  	public double getRemainingCapacity() {
52  		return this.remainingCapacity;
53  	}
54  
55  	@Override
56  	public double getTimeRemaining() {
57  		return this.timeRemaining;
58  	}
59  
60  	/**
61  	 * Battery Information
62  	 */
63  	public static PowerSource[] getPowerSources() {
64  		// Windows provides a single unnamed battery
65  		String name = "System Battery";
66  		WindowsPowerSource[] psArray = new WindowsPowerSource[1];
67  		// Get structure
68  		SystemBatteryState batteryState = new SystemBatteryState();
69  		if (0 != PowrProf.INSTANCE.CallNtPowerInformation(
70  				PowrProf.SYSTEM_BATTERY_STATE, null, new NativeLong(0),
71  				batteryState, new NativeLong(batteryState.size()))
72  				|| batteryState.batteryPresent == 0) {
73  			psArray[0] = new WindowsPowerSource("Unknown", 0d, -1d);
74  		} else {
75  			int estimatedTime = -2; // -1 = unknown, -2 = unlimited
76  			if (batteryState.acOnLine == 0 && batteryState.charging == 0
77  					&& batteryState.discharging > 0)
78  				estimatedTime = batteryState.estimatedTime;
79  			long maxCapacity = FormatUtil
80  					.getUnsignedInt(batteryState.maxCapacity);
81  			long remainingCapacity = FormatUtil
82  					.getUnsignedInt(batteryState.remainingCapacity);
83  
84  			psArray[0] = new WindowsPowerSource(name,
85  					(double) remainingCapacity / maxCapacity,
86  					estimatedTime);
87  		}
88  
89  		return psArray;
90  	}
91  }
