17  package oshi.software.os.windows.nt;
18  
19  import com.sun.jna.Native;
20  import com.sun.jna.platform.win32.WinBase;
21  
22  /**
23   * Windows Kernel32
24   * 
25   * @author widdis[at]gmail[dot]com
26   */
27  public interface Kernel32 extends com.sun.jna.platform.win32.Kernel32 {
28  	Kernel32 INSTANCE = (Kernel32) Native.loadLibrary("Kernel32", Kernel32.class);
29  
30  	public static final long WIN32_TIME_OFFSET = (369L * 365L + 89L) * 24L * 3600L * 1000L;
31  
32  	// TODO: Submit this change to JNA Kernel32 class
33  	/**
34  	 * Retrieves system timing information. On a multiprocessor system, the
35  	 * values returned are the sum of the designated times across all
36  	 * processors.
37  	 * 
38  	 * Times are offset by {@value #WIN32_TIME_OFFSET}.
39  	 * 
40  	 * @param lpIdleTime
41  	 *            A pointer to a FILETIME structure that receives the amount of
42  	 *            time that the system has been idle.
43  	 * @param lpKernelTime
44  	 *            A pointer to a FILETIME structure that receives the amount of
45  	 *            time that the system has spent executing in Kernel mode
46  	 *            (including all threads in all processes, on all processors).
47  	 *            This time value also includes the amount of time the system
48  	 *            has been idle.
49  	 * @param lpUserTime
50  	 *            A pointer to a FILETIME structure that receives the amount of
51  	 *            time that the system has spent executing in User mode
52  	 *            (including all threads in all processes, on all processors).
53  	 * @return If the function succeeds, the return value is nonzero. If the
54  	 *         function fails, the return value is zero and errno is set.
55  	 */
56  	public int GetSystemTimes(WinBase.FILETIME lpIdleTime, WinBase.FILETIME lpKernelTime, WinBase.FILETIME lpUserTime);
57  
58  }
