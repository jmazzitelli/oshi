17  package oshi.software.os.windows.nt;
18  
19  import java.io.File;
20  import java.util.ArrayList;
21  import java.util.List;
22  
23  import javax.swing.filechooser.FileSystemView;
24  
25  import oshi.software.os.OSFileStore;
26  
27  /**
28   * The Windows File System contains {@link OSFileStore}s which are a storage
29   * pool, device, partition, volume, concrete file system or other implementation
30   * specific means of file storage. In Windows, these are represented by a drive
31   * letter, e.g., "A:\" and "C:\"
32   * 
33   * @author widdis[at]gmail[dot]com
34   */
35  public class WindowsFileSystem {
36  	/**
37  	 * Gets File System Information.
38  	 * 
39  	 * @return An array of {@link OSFileStore} objects representing mounted
40  	 *         volumes. May return disconnected volumes with
41  	 *         {@link OSFileStore#getTotalSpace()} = 0.
42  	 */
43  	public static OSFileStore[] getFileStores() {
44  		// File.listRoots() has more information for Windows
45  		// than FileSystem.getDefalut().getFileStores()
46  		File[] roots = File.listRoots();
47  		FileSystemView fsv = FileSystemView.getFileSystemView();
48  		List<OSFileStore> fsList = new ArrayList<>();
49  		for (File f : roots) {
50  			fsList.add(new OSFileStore(fsv.getSystemDisplayName(f), fsv
51  					.getSystemTypeDescription(f), f.getUsableSpace(), f
52  					.getTotalSpace()));
53  		}
54  		return fsList.toArray(new OSFileStore[fsList.size()]);
55  	}
56  }
