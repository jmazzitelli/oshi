17  package oshi.software.os.linux;
18  
19  import java.util.Arrays;
20  import java.util.List;
21  
22  import com.sun.jna.Library;
23  import com.sun.jna.Native;
24  import com.sun.jna.NativeLong;
25  import com.sun.jna.Structure;
26  
27  public interface Libc extends Library {
28  
29  	public static final Libc INSTANCE = (Libc) Native.loadLibrary("c",
30  			Libc.class);
31  
32  	public static final class Sysinfo extends Structure {
33  		public NativeLong uptime; // Seconds since boot
34  		// 1, 5, and 15 minute load averages
35  		public NativeLong[] loads = new NativeLong[3];
36  		public NativeLong totalram; // Total usable main memory size
37  		public NativeLong freeram; // Available memory size
38  		public NativeLong sharedram; // Amount of shared memory
39  		public NativeLong bufferram; // Memory used by buffers
40  		public NativeLong totalswap; // Total swap space size
41  		public NativeLong freeswap; // swap space still available
42  		public short procs; // Number of current processes
43  		public NativeLong totalhigh; // Total high memory size
44  		public NativeLong freehigh; // Available high memory size
45  		public int mem_unit; // Memory unit size in bytes
46  		public byte[] _f = new byte[8]; // Won't be written for 64-bit systems
47  
48  		@Override
49  		protected List getFieldOrder() {
50  			return Arrays.asList(new String[] { "uptime", "loads", "totalram",
51  					"freeram", "sharedram", "bufferram", "totalswap",
52  					"freeswap", "procs", "totalhigh", "freehigh", "mem_unit",
53  					"_f" });
54  		}
55  	}
56  
57  	int sysinfo(Sysinfo info);
58  
59  }
