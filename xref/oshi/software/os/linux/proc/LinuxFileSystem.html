17  package oshi.software.os.linux.proc;
18  
19  import java.io.IOException;
20  import java.nio.file.FileStore;
21  import java.nio.file.FileSystems;
22  import java.util.ArrayList;
23  import java.util.List;
24  
25  import oshi.software.os.OSFileStore;
26  
27  /**
28   * The Mac File System contains {@link OSFileStore}s which are a storage pool,
29   * device, partition, volume, concrete file system or other implementation
30   * specific means of file storage. In Linux, these are found in the /proc/mount
31   * filesystem, excluding temporary and kernel mounts.
32   * 
33   * @author widdis[at]gmail[dot]com
34   */
35  public class LinuxFileSystem {
36  	/**
37  	 * Gets File System Information.
38  	 * 
39  	 * @return An array of {@link FileStore} objects representing mounted
40  	 *         volumes. May return disconnected volumes with
41  	 *         {@link OSFileStore#getTotalSpace()} = 0.
42  	 */
43  	public static OSFileStore[] getFileStores() {
44  		List<OSFileStore> fsList = new ArrayList<>();
45  		for (FileStore store : FileSystems.getDefault().getFileStores()) {
46  			// FileStore toString starts with path, then a space, then name in
47  			// parentheses e.g., "/ (/dev/sda1)" and "/proc (proc)"
48  			String path = store.toString().replace(" (" + store.name() + ")", "");
49  			// Exclude special directories
50  			if (path.startsWith("/proc") || path.startsWith("/sys") || path.startsWith("/run") || path.equals("/dev")
51  					|| path.equals("/dev/pts"))
52  				continue;
53  			String name = store.name();
54  			if (path.equals("/"))
55  				name = "/";
56  			String description = "Mount Point";
57  			if (store.name().startsWith("/dev"))
58  				description = "Local Disk";
59  			try {
60  				fsList.add(new OSFileStore(name, description, store.getUsableSpace(), store.getTotalSpace()));
61  			} catch (IOException e) {
62  				// get*Space() may fail for ejected CD-ROM, etc.
63  				continue;
64  			}
65  		}
66  		return fsList.toArray(new OSFileStore[fsList.size()]);
67  	}
68  }
