96  	public static class VMStatistics extends Structure {
97  		public int free_count; // # of pages free
98  		public int active_count; // # of pages active
99  		public int inactive_count; // # of pages inactive
100 		public int wire_count; // # of pages wired down
101 		public int zero_fill_count; // # of zero fill pages
102 		public int reactivations; // # of pages reactivated
103 		public int pageins; // # of pageins
104 		public int pageouts; // # of pageouts
105 		public int faults; // # of faults
106 		public int cow_faults; // # of copy-on-writes
107 		public int lookups; // object cache lookups
108 		public int hits; // object cache hits
109 		public int purgeable_count; // # of pages purgeable
110 		public int purges; // # of pages purged
111 		// # of pages speculative (included in free_count)
112 		public int speculative_count;
113 
114 		@Override
115 		protected List getFieldOrder() {
116 			return Arrays.asList(new String[] { "free_count", "active_count",
117 					"inactive_count", "wire_count", "zero_fill_count",
118 					"reactivations", "pageins", "pageouts", "faults",
119 					"cow_faults", "lookups", "hits", "purgeable_count",
120 					"purges", "speculative_count" });
121 		}
122 	}

124 	public static class VMStatistics64 extends Structure {
125 		public int free_count; // # of pages free
126 		public int active_count; // # of pages active
127 		public int inactive_count; // # of pages inactive
128 		public int wire_count; // # of pages wired down
129 		public long zero_fill_count; // # of zero fill pages
130 		public long reactivations; // # of pages reactivated
131 		public long pageins; // # of pageins
132 		public long pageouts; // # of pageouts
133 		public long faults; // # of faults
134 		public long cow_faults; // # of copy-on-writes
135 		public long lookups; // object cache lookups
136 		public long hits; // object cache hits
137 		public long purges; // # of pages purged
138 		public int purgeable_count; // # of pages purgeable
139 		// # of pages speculative (included in free_count)
140 		public int speculative_count;
141 		public long decompressions; // # of pages decompressed
142 		public long compressions; // # of pages compressed
143 		// # of pages swapped in (via compression segments)
144 		public long swapins;
145 		// # of pages swapped out (via compression segments)
146 		public long swapouts;
147 		// # of pages used by the compressed pager to hold all the
148 		// compressed data
149 		public int compressor_page_count;
150 		public int throttled_count; // # of pages throttled
151 		// # of pages that are file-backed (non-swap)
152 		public int external_page_count;
153 		public int internal_page_count; // # of pages that are anonymous
154 		// # of pages (uncompressed) held within the compressor.
155 		public long total_uncompressed_pages_in_compressor;
156 
157 		@Override
158 		protected List getFieldOrder() {
159 			return Arrays.asList(new String[] { "free_count", "active_count",
160 					"inactive_count", "wire_count", "zero_fill_count",
161 					"reactivations", "pageins", "pageouts", "faults",
162 					"cow_faults", "lookups", "hits", "purges",
163 					"purgeable_count", "speculative_count", "decompressions",
164 					"compressions", "swapins", "swapouts",
165 					"compressor_page_count", "throttled_count",
166 					"external_page_count", "internal_page_count",
167 					"total_uncompressed_pages_in_compressor" });
168 		}
169 	}
