17  package oshi.hardware;
18  
19  /**
20   * The Central Processing Unit (CPU) or the processor is the portion of a
21   * computer system that carries out the instructions of a computer program, and
22   * is the primary element carrying out the computer's functions.
23   * 
24   * @author dblock[at]dblock[dot]org
25   */
26  public interface Processor {
27  	/**
28  	 * Gets the processor number of this object, passed as an arg in the
29  	 * constructor.
30  	 * 
31  	 * @return The processor number
32  	 */
33  	int getProcessorNumber();
34  
35  	/**
36  	 * Processor vendor.
37  	 * 
38  	 * @return String.
39  	 */
40  	String getVendor();
41  
42  	/**
43  	 * Set processor vendor.
44  	 * 
45  	 * @param vendor
46  	 *            Vendor.
47  	 */
48  	void setVendor(String vendor);
49  
50  	/**
51  	 * Name, eg. Intel(R) Core(TM)2 Duo CPU T7300 @ 2.00GHz
52  	 * 
53  	 * @return Processor name.
54  	 */
55  	String getName();
56  
57  	/**
58  	 * Set processor name.
59  	 * 
60  	 * @param name
61  	 *            Name.
62  	 */
63  	void setName(String name);
64  
65  	/**
66  	 * Vendor frequency (in Hz).
67  	 * 
68  	 * @return Processor frequency.
69  	 */
70  	long getVendorFreq();
71  
72  	/**
73  	 * Set processor vendor frequency (in Hz).
74  	 * 
75  	 * @param freq
76  	 *            Frequency.
77  	 */
78  	void setVendorFreq(long freq);
79  
80  	/**
81  	 * Identifier, eg. x86 Family 6 Model 15 Stepping 10.
82  	 * 
83  	 * @return Processor identifier.
84  	 */
85  	String getIdentifier();
86  
87  	/**
88  	 * Set processor identifier.
89  	 * 
90  	 * @param identifier
91  	 *            Identifier.
92  	 */
93  	void setIdentifier(String identifier);
94  
95  	/**
96  	 * Is CPU 64bit?
97  	 * 
98  	 * @return True if cpu is 64bit.
99  	 */
100 	boolean isCpu64bit();
101 
102 	/**
103 	 * Set flag is cpu is 64bit.
104 	 * 
105 	 * @param cpu64
106 	 *            True if cpu is 64.
107 	 */
108 	void setCpu64(boolean cpu64);
109 
110 	/**
111 	 * @return the _stepping
112 	 */
113 	String getStepping();
114 
115 	/**
116 	 * @param _stepping
117 	 *            the _stepping to set
118 	 */
119 	void setStepping(String _stepping);
120 
121 	/**
122 	 * @return the _model
123 	 */
124 	String getModel();
125 
126 	/**
127 	 * @param _model
128 	 *            the _model to set
129 	 */
130 	void setModel(String _model);
131 
132 	/**
133 	 * @return the _family
134 	 */
135 	String getFamily();
136 
137 	/**
138 	 * @param _family
139 	 *            the _family to set
140 	 */
141 	void setFamily(String _family);
142 
143 	/**
144 	 * Returns the "recent cpu usage" for the whole system by counting ticks
145 	 * from {@link #getSystemCpuLoadTicks()} between successive calls of this
146 	 * method, with a minimum interval slightly less than 1 second.
147 	 * 
148 	 * If less than one second has elapsed since the last call of this method,
149 	 * it will return a calculation based on the tick counts and times of the
150 	 * previous two calls. If at least a second has elapsed, it will return the
151 	 * average CPU load for the interval and update the "last called" times.
152 	 * 
153 	 * This method is intended to be used for periodic polling at intervals of 1
154 	 * second or longer.
155 	 * 
156 	 * @return CPU load in %
157 	 * 
158 	 * @deprecated Replaced in 1.3 by {@link #getSystemCpuLoadBetweenTicks()}
159 	 */
160 	@Deprecated
161 	float getLoad();
162 
163 	/**
164 	 * Returns the "recent cpu usage" for the whole system by counting ticks
165 	 * from {@link #getSystemCpuLoadTicks()} between successive calls of this
166 	 * method, with a minimum interval slightly less than 1 second.
167 	 * 
168 	 * If less than one second has elapsed since the last call of this method,
169 	 * it will return a calculation based on the tick counts and times of the
170 	 * previous two calls. If at least a second has elapsed, it will return the
171 	 * average CPU load for the interval and update the "last called" times.
172 	 * 
173 	 * This method is intended to be used for periodic polling at intervals of 1
174 	 * second or longer.
175 	 * 
176 	 * @return CPU load between 0 and 1 (100%)
177 	 */
178 	double getSystemCpuLoadBetweenTicks();
179 
180 	/**
181 	 * Get System-wide CPU Load tick counters. Returns an array with four
182 	 * elements representing clock ticks or milliseconds (platform dependent)
183 	 * spent in User (0), Nice (1), System (2), and Idle (3) states. By
184 	 * measuring the difference between ticks across a time interval, CPU load
185 	 * over that interval may be calculated.
186 	 * 
187 	 * @return An array of 4 long values representing time spent in User,
188 	 *         Nice(if applicable), System, and Idle states.
189 	 */
190 	long[] getSystemCpuLoadTicks();
191 
192 	/**
193 	 * Returns the "recent cpu usage" for the whole system from
194 	 * {@link com.sun.management.OperatingSystemMXBean#getSystemCpuLoad()} if a
195 	 * user is running the Oracle JVM. This value is a double in the [0.0,1.0]
196 	 * interval. A value of 0.0 means that all CPUs were idle during the recent
197 	 * period of time observed, while a value of 1.0 means that all CPUs were
198 	 * actively running 100% of the time during the recent period being
199 	 * observed. All values between 0.0 and 1.0 are possible depending of the
200 	 * activities going on in the system. If the system recent cpu usage is not
201 	 * available, the method returns a negative value.
202 	 * 
203 	 * Calling this method immediately upon instantiating the {@link Processor}
204 	 * may give unreliable results.
205 	 * 
206 	 * If a user is not running the Oracle JVM, this method will default to the
207 	 * behavior and return value of {@link #getSystemCpuLoadBetweenTicks()}.
208 	 * 
209 	 * @return the "recent cpu usage" for the whole system; a negative value if
210 	 *         not available.
211 	 */
212 	@SuppressWarnings("restriction")
213 	double getSystemCpuLoad();
214 
215 	/**
216 	 * Returns the system load average for the last minute from
217 	 * {@link java.lang.management.OperatingSystemMXBean#getSystemLoadAverage()}
218 	 * . The system load average is the sum of the number of runnable entities
219 	 * queued to the available processors and the number of runnable entities
220 	 * running on the available processors averaged over a period of time. The
221 	 * way in which the load average is calculated is operating system specific
222 	 * but is typically a damped time-dependent average.
223 	 * 
224 	 * If the load average is not available, a negative value is returned.
225 	 * 
226 	 * This method is designed to provide a hint about the system load and may
227 	 * be queried frequently. The load average may be unavailable on some
228 	 * platforms (e.g., Windows) where it is expensive to implement this method.
229 	 * 
230 	 * @return the system load average; or a negative value if not available.
231 	 */
232 	double getSystemLoadAverage();
233 
234 	/**
235 	 * Returns the "recent cpu usage" for this processor by counting ticks for
236 	 * this processor from {@link #getProcessorCpuLoadTicks()} between
237 	 * successive calls of this method, with a minimum interval slightly less
238 	 * than 1 second.
239 	 * 
240 	 * If less than one second has elapsed since the last call of this method,
241 	 * it will return a calculation based on the tick counts and times of the
242 	 * previous two calls. If at least a second has elapsed, it will return the
243 	 * average CPU load for the interval and update the "last called" times.
244 	 * 
245 	 * This method is intended to be used for periodic polling (iterating over
246 	 * all processors) at intervals of 1 second or longer.
247 	 * 
248 	 * @return CPU load between 0 and 1 (100%)
249 	 */
250 
251 	double getProcessorCpuLoadBetweenTicks();
252 
253 	/**
254 	 * Get this Processor's CPU Load tick counters. Returns an array with four
255 	 * elements representing clock ticks or milliseconds (platform dependent)
256 	 * spent in User (0), Nice (1), System (2), and Idle (3) states. By
257 	 * measuring the difference between ticks across a time interval, CPU load
258 	 * over that interval may be calculated.
259 	 * 
260 	 * @return An array of 4 long values representing time spent in User,
261 	 *         Nice(if applicable), System, and Idle states.
262 	 */
263 	long[] getProcessorCpuLoadTicks();
264 
265 	/**
266 	 * Get the System uptime (time since boot).
267 	 * 
268 	 * @return Number of seconds since boot.
269 	 */
270 	long getSystemUptime();
271 }
