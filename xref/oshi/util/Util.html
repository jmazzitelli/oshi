17  package oshi.util;
18  
19  import org.slf4j.Logger;
20  import org.slf4j.LoggerFactory;
21  
22  /**
23   * General utility methods
24   * 
25   * @author widdis[at]gmail[dot]com
26   */
27  public class Util {
28  	private static final Logger LOG = LoggerFactory.getLogger(Util.class);
29  
30  	/**
31  	 * Sleeps for the specified number of milliseconds
32  	 * 
33  	 * @param ms
34  	 *            How long to sleep
35  	 */
36  	public static void sleep(long ms) {
37  		try {
38  			LOG.trace("Sleeping for {} ms", ms);
39  			Thread.sleep(ms);
40  		} catch (InterruptedException e) {
41  			LOG.warn("Interrupted while sleeping for {} ms", ms);
42  		}
43  	}
44  
45  	/**
46  	 * Sleeps for the specified number of milliseconds after the given system
47  	 * time in milliseconds. If that number of milliseconds has already elapsed,
48  	 * does nothing.
49  	 * 
50  	 * @param startTime
51  	 *            System time in milliseconds to sleep after
52  	 * @param ms
53  	 *            How long after startTime to sleep
54  	 */
55  	public static void sleepAfter(long startTime, long ms) {
56  		long now = System.currentTimeMillis();
57  		long until = startTime + ms;
58  		LOG.trace("Sleeping until {}", until);
59  		if (now < until)
60  			sleep(until - now);
61  	}
62  }
