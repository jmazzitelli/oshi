17  package oshi.util;
18  
19  import java.math.BigDecimal;
20  
21  /**
22   * Formatting utility for appending units or converting between number types.
23   * 
24   * @author dblock[at]dblock[dot]org
25   */
26  public abstract class FormatUtil {
27  
28  	/**
29  	 * Added these variable for easier reading Using the IEC Standard for naming
30  	 * the units
31  	 * (http://en.wikipedia.org/wiki/International_Electrotechnical_Commission)
32  	 */
33  	final private static long kibiByte = 1024L;
34  	final private static long mebiByte = kibiByte * kibiByte;
35  	final private static long gibiByte = mebiByte * kibiByte;
36  	final private static long tebiByte = gibiByte * kibiByte;
37  	final private static long pebiByte = tebiByte * kibiByte;
38  	final private static long exbiByte = pebiByte * kibiByte;
39  
40  	/**
41  	 * Hertz related variables
42  	 */
43  	final private static long kiloHertz = 1000L;
44  	final private static long megaHertz = kiloHertz * kiloHertz;
45  	final private static long gigaHertz = megaHertz * kiloHertz;
46  	final private static long teraHertz = gigaHertz * kiloHertz;
47  	final private static long petaHertz = teraHertz * kiloHertz;
48  	final private static long exaHertz = petaHertz * kiloHertz;
49  
50  	/**
51  	 * Format bytes into a string to a rounded string representation. Using the
52  	 * JEDEC representation for KB, MB and GB Using the IEC representation for
53  	 * TiB and PiB
54  	 * 
55  	 * @param bytes
56  	 *            Bytes.
57  	 * @return Rounded string representation of the byte size.
58  	 */
59  	public static String formatBytes(long bytes) {
60  		if (bytes == 1) { // bytes
61  			return String.format("%d byte", bytes);
62  		} else if (bytes < kibiByte) { // bytes
63  			return String.format("%d bytes", bytes);
64  		} else if (bytes < mebiByte && bytes % kibiByte == 0) { // KB
65  			return String.format("%.0f KB", (double) bytes / kibiByte);
66  		} else if (bytes < mebiByte) { // KB
67  			return String.format("%.1f KB", (double) bytes / kibiByte);
68  		} else if (bytes < gibiByte && bytes % mebiByte == 0) { // MB
69  			return String.format("%.0f MB", (double) bytes / mebiByte);
70  		} else if (bytes < gibiByte) { // MB
71  			return String.format("%.1f MB", (double) bytes / mebiByte);
72  		} else if (bytes % gibiByte == 0 && bytes < tebiByte) { // GB
73  			return String.format("%.0f GB", (double) bytes / gibiByte);
74  		} else if (bytes < tebiByte) { // GB
75  			return String.format("%.1f GB", (double) bytes / gibiByte);
76  		} else if (bytes % tebiByte == 0 && bytes < pebiByte) { // TiB
77  			return String.format("%.0f TiB", (double) bytes / tebiByte);
78  		} else if (bytes < pebiByte) { // TiB
79  			return String.format("%.1f TiB", (double) bytes / tebiByte);
80  		} else if (bytes % pebiByte == 0 && bytes < exbiByte) { // PiB
81  			return String.format("%.0f PiB", (double) bytes / pebiByte);
82  		} else if (bytes < exbiByte) { // PiB
83  			return String.format("%.1f PiB", (double) bytes / pebiByte);
84  		} else {
85  			return String.format("%d bytes", bytes);
86  		}
87  	}
88  
89  	/**
90  	 * Format hertz into a string to a rounded string representation.
91  	 * 
92  	 * @param hertz
93  	 *            Hertz.
94  	 * @return Rounded string representation of the hertz size.
95  	 */
96  	public static String formatHertz(long hertz) {
97  		if (hertz < kiloHertz) { // Hz
98  			return String.format("%d Hz", hertz);
99  		} else if (hertz < megaHertz && hertz % kiloHertz == 0) { // KHz
100 			return String.format("%.0f kHz", (double) hertz / kiloHertz);
101 		} else if (hertz < megaHertz) {
102 			return String.format("%.1f kHz", (double) hertz / kiloHertz);
103 		} else if (hertz < gigaHertz && hertz % megaHertz == 0) { // MHz
104 			return String.format("%.0f MHz", (double) hertz / megaHertz);
105 		} else if (hertz < gigaHertz) {
106 			return String.format("%.1f MHz", (double) hertz / megaHertz);
107 		} else if (hertz < teraHertz && hertz % gigaHertz == 0) { // GHz
108 			return String.format("%.0f GHz", (double) hertz / gigaHertz);
109 		} else if (hertz < teraHertz) {
110 			return String.format("%.1f GHz", (double) hertz / gigaHertz);
111 		} else if (hertz < petaHertz && hertz % teraHertz == 0) { // THz
112 			return String.format("%.0f THz", (double) hertz / teraHertz);
113 		} else if (hertz < petaHertz) {
114 			return String.format("%.1f THz", (double) hertz / teraHertz);
115 		} else if (hertz < exaHertz && hertz % petaHertz == 0) { // EHz
116 			return String.format("%.0f EHz", (double) hertz / petaHertz);
117 		} else if (hertz < exaHertz) {
118 			return String.format("%.1f EHz", (double) hertz / petaHertz);
119 		} else {
120 			return String.format("%d Hz", hertz);
121 		}
122 	}
123 
124 	/**
125 	 * Round to certain number of decimals
126 	 *
127 	 * @param d
128 	 * @param decimalPlace
129 	 * @return rounded result
130 	 */
131 	public static float round(float d, int decimalPlace) {
132 		BigDecimal bd = new BigDecimal(Float.toString(d));
133 		bd = bd.setScale(decimalPlace, BigDecimal.ROUND_HALF_UP);
134 		return bd.floatValue();
135 	}
136 
137 	/**
138 	 * Convert unsigned int to signed long
139 	 * 
140 	 * @param x
141 	 * @return long value of x unsigned
142 	 */
143 	public static long getUnsignedInt(int x) {
144 		return x & 0x00000000ffffffffL;
145 	}
146 }
