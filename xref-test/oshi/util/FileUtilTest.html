17  package oshi.util;
18  
19  import static org.junit.Assert.assertEquals;
20  
21  import java.io.IOException;
22  import java.util.List;
23  
24  import org.junit.Test;
25  
26  /**
27   * The Class FileUtilTest.
28   */
29  public class FileUtilTest {
30  
31  	/** The thisclass. */
32  	private static String THISCLASS = "src/test/java/oshi/util/FileUtilTest.java";
33  
34  	/**
35  	 * Test read file.
36  	 */
37  	@Test
38  	public void testReadFile() {
39  		List<String> thisFile = null;
40  		try {
41  			thisFile = FileUtil.readFile(THISCLASS);
42  		} catch (IOException e) {
43  			e.printStackTrace();
44  			return;
45  		}
46  		// Comment ONE line
47  		int lineOne = 0;
48  		// Comment TWO line
49  		int lineTwo = 0;
50  		for (int i = 0; i < thisFile.size(); i++) {
51  			String line = thisFile.get(i);
52  			if (line.contains("Comment ONE line"))
53  				lineOne = i;
54  			if (line.contains("Comment TWO line"))
55  				lineTwo = i;
56  		}
57  		assertEquals(2, lineTwo - lineOne);
58  	}
59  }
